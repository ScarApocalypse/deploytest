        # 双飞翼布局

        <!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
    <style>
      body {
        min-width: 500px;
      }

      #container {
        width: 100%;
      }

      .column {
        float: left;
      }

      #center {
        margin-left: 200px;
        margin-right: 150px;
      }

      #left {
        width: 200px;
        margin-left: -100%;
      }

      #right {
        width: 150px;
        margin-left: -150px;
      }

      #footer {
        clear: both;
      }

      #header,
      #footer {
        background-color: #ccc;
        height: 100px;
      }

      #left,
      #right {
        height: 600px;
        background-color: #eee;
      }

      #center {
        height: 100px;
        background-color: skyblue;
      }
    </style>
  </head>
  <body>
    <div id="header"></div>
    <div id="container" class="column">
      <div id="center"></div>
    </div>
    <div id="left" class="column"></div>
    <div id="right" class="column"></div>
    <div id="footer"></div>
  </body>
</html>
