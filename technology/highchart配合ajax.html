  • highchart配合ajax

    hignchart官网原文教程

    • 前端代码
    <!DOCTYPE html>
<html>
  <head>
    <meta charset="utf-8" />
    <title>第一个 Highcharts 图表</title>
  </head>
  <body>
    <script src="https://cdn.bootcdn.net/ajax/libs/jquery/3.5.1/jquery.min.js"></script>
    <!-- 图表容器 DOM -->
    <div id="container" style="width: 600px; height: 400px;"></div>
    <!-- 引入 highcharts.js -->
    <script src="http://cdn.highcharts.com.cn/highcharts/highcharts.js"></script>

    <script>
      var chart = null; // 定义全局变量
      /**
       * Ajax 请求数据接口，并通过 Highcharts 提供的函数进行动态更新
       * 接口调用完毕后间隔 1 s 继续调用本函数，以达到实时请求数据，实时更新的效果
       */
      function requestData() {
        $.ajax({
          url: "live-server-data.php",
          success: function (point) {
            var series = chart.series[0],
              shift = series.data.length > 20; // 当数据点数量超过 20 个，则指定删除第一个点

            // 新增点操作
            //具体的参数详见：https://api.hcharts.cn/highcharts#Series.addPoint
            chart.series[0].addPoint(point, true, shift);

            // 一秒后继续调用本函数
            setTimeout(requestData, 500);
          },
          cache: false,
        });
      }
      $(document).ready(function () {
        chart = Highcharts.chart("container", {
          chart: {
            type: "spline",
            events: {
              load: requestData, // 图表加载完毕后执行的回调函数
            },
          },
          title: {
            text: "Live random data",
          },
          xAxis: {
            type: "datetime",
            tickPixelInterval: 150,
            maxZoom: 20 * 1000,
          },
          yAxis: {
            minPadding: 0.2,
            maxPadding: 0.2,
            title: {
              text: "Value",
              margin: 80,
            },
          },
          series: [
            {
              name: "随机数据",
              data: [],
            },
          ],
          credits:{
            enabled: false // 禁用版权信息
          }
        });
      });
    </script>
  </body>
</html>

    <?php
// 指定返回格式为 JSON
header("Content-type: text/json");

// 获取当前时间，PHP 时间戳是秒为单位的，JS 中则是毫秒，所以这里乘以 1000
$x = time() * 1000;
// y 值为随机值
$y = rand(0, 100);

// 创建 PHP 数组，并最终用 json_encode 转换成 JSON 字符串
$ret = array($x, $y);
echo json_encode($ret);
